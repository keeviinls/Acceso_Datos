16       public static void guardarObjeto(File file) throws IOException {
17           // CREAMOS LA PERSONA Y LE AÑADIMOS LOS DATOS
18           Persona persona = new Persona(1, "Zoro", 1, "49998728V");
19           System.out.println("La persona ha sido creada");
20   
21               // CREAMOS UN FLUJO DE SALIDA, YA QUE HAY QUE ENVIAR DATOS DESDE UN PROGRAMA HACIA UN DESTINO
22               FileOutputStream fichero_salida = new FileOutputStream(file);
23   
24               // CREAR EL OBJECTOUTPUTSTREAM QUE ESTE ASOCIADO AL FLUJO
25               ObjectOutputStream out = new ObjectOutputStream(fichero_salida);
26   
27               // DENTRO DEL FLUJO DE SALIDA, ESCRIBIR A LA "PERSONA""
28               out.writeObject(persona);
29   
30               // CERRAR EL OBJETO DE FLUJO Y EL FICHERO
31               out.close();
32               fichero_salida.close();

36       public static void recuperarObjeto(File file) throws IOException, ClassNotFoundException {
37   
38               // CREACION DEL FLUJO DE ENTRADA, YA QUE HAY QUE ENVIAR DATOS DESDE UNA FUENTA A UN PROGRAMA
39               FileInputStream fichero_entrada = new FileInputStream(file);
40   
41               // CREAR EL OBJECTOUTPUTSTREAM QUE ESTE ASOCIADO AL FLUJO
42               ObjectInputStream in = new ObjectInputStream(fichero_entrada);
43   
44               // Lectura del objeto de tipo Persona desde el flujo de entrada y casting al tipo Persona
45               Persona persona = (Persona) in.readObject();
46   
47               // Modificar las propiedades del objeto Persona si es necesario
48               persona.setEdad(31);
49   
50               // Creación de un nuevo flujo de salida de bytes hacia el archivo
51               FileOutputStream fileOut = new FileOutputStream(file);
52               // Creación de un nuevo ObjectOutputStream que se conecta al nuevo flujo de salida de bytes
53               ObjectOutputStream out = new ObjectOutputStream(fileOut);
54               // Escribir el objeto persona modificado en el nuevo flujo de salida
55               out.writeObject(persona);
56   
57               // Cerrar el ObjectInputStream, el flujo de entrada de bytes, el ObjectOutputStream y el flujo de salida de bytes
58               in.close();
59               fichero_entrada.close();
60               out.close();
61               fileOut.close();
62   
63   
64   
65       }
